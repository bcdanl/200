2 Statistical Transformation

ggplot(diamonds,
       aes(x = cut)) +
  geom_bar()

# to make a simple data.frame
demo <- tribble(         
  ~cut,         ~freq,
  "Fair",       1610,
  "Good",       4906,
  "Very Good",  12082,
  "Premium",    13791,
  "Ideal",      21551 )


ggplot(data = demo,
       aes(x = cut)) +
  geom_bar()

ggplot(data = demo) +
  geom_bar(mapping = 
             aes(x = cut, 
                 y = freq), 
           stat = "identity")

diamonds <- diamonds
ggplot(diamonds,
       aes(x = cut, fill = cut)) +
  geom_bar()

ggplot(diamonds,
       aes(x = cut, fill = clarity)) +
  geom_bar()
