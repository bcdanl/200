Logical Operators

(flights$month == 1) & (flights$day == 1)
(flights$month == 2) & (flights$day == 1)


(flights$month == 2) | (flights$day == 1)



(flights$month == 1)
!(flights$month == 1)
df <- data.frame(
  num = c(8, 9, 10, 11),
  chr = c("A", "C", "B", "A"))
